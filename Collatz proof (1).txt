# Golden Ratio Ideals for C, H, F block frequencies
PHI = (1 + math.sqrt(5)) / 2
IDEAL_RATIOS = {
    "F": 1 / PHI**2,  # ~0.382 (Stall/Hangup)
    "C": 1 / PHI,     # ~0.618 (Strong Drop)
    # Note: The H-block (Mild Descent) is hypothesized to be a composite of
    # C and F behaviors, so we primarily watch C and F converge.
}


def classify_collatz_block(n):
    """
    Classifies a number `n` (which is the result of a 3n+1 operation)
    into a C, H, or F block based on the subsequent number of n/2 divisions.
    The number of divisions is equal to the number of trailing zeros in the
    binary representation of the even number.
    """
    if n % 2 != 0:
        return None # Should only be called on an even number


    # (n & -n) isolates the lowest set bit (e.g., for 12 (1100), result is 4 (0100))
    # .bit_length() - 1 gives the number of trailing zeros (e.g., for 4, it's 2)
    num_divisions = (n & -n).bit_length() - 1


    # These thresholds are based on the conceptual model:
    # F (Flat): A stall or minimal drop (only one n/2 division)
    # C (Cascade): A strong, efficient drop (more than 2 n/2 divisions)
    # H (High-k): A mild, balanced descent (exactly 2 n/2 divisions)
    if num_divisions == 1:
        return "F"  # Flat, Stall/Hangup
    elif num_divisions > 2:
        return "C"  # Cascade, Strong Drop
    else: # num_divisions == 2
        return "H"  # High-k, Mild Descent

        n = start_n

            if n % 2 == 0:

                c_ratio = block_counts['C'] / total_blocks

                print(f"  Deviation from Ideal: C={abs(c_ratio - IDEAL_RATIOS['C']):.4f}, F={abs(f_ratio - IDEAL_RATIOS['F']):.4f}")


    print("\n" + "="*80)

    return block_counts

import math
import random


class PrimePredictionEngine:
    """
    Predicts prime numbers by traversing the Collatz harmonic lattice.
    """
    def __init__(self, start_prime=3, max_prime=10**10):
        self.last_known_prime = start_prime
        self.max_prime = max_prime
        self.primes_found = [2, 3]
        self.candidates_tested = 0


    def _is_prime(self, n, k=5):
        """
        Uses the Miller-Rabin primality test, efficient for large numbers.
        """
        if n < 2: return False
        if n == 2 or n == 3: return True
        if n % 2 == 0 or n % 3 == 0: return False
        d, s = n - 1, 0
        while d % 2 == 0:
            d //= 2
            s += 1
        for _ in range(k):
            a = random.randint(2, n - 2)
            x = pow(a, d, n)
            if x == 1 or x == n - 1:
                continue
            for _ in range(s - 1):
                x = pow(x, 2, n)
                if x == n - 1:
                    break
            else:
                return False
        return True


    def _classify_collatz_block(self, n):
        """
        Classifies a number `n` (the result of a 3n+1 operation)
        into a C, H, or F block.
        """
        if n % 2 != 0: return None
        num_divisions = (n & -n).bit_length() - 1
        if num_divisions == 1:
            return "F"  # Flat, Stall/Hangup (The "Push" / Expansion Crest)
        elif num_divisions > 2:
            return "C"  # Cascade, Strong Drop
        else:
            return "H"  # High-k, Mild Descent


    def _phase_hop(self, start_num):
        """
        Performs a "phase-hop" by searching for the next number that
        produces a desirable (F-block) crest in the Collatz lattice.
        This is the core of the predictive model.
        """
        num = start_num
        while True:
            num += 2 # Only need to check odd numbers
            self.candidates_tested += 1
            
            # A "crest" is a 3n+1 operation.
            crest_val = 3 * num + 1
            block_type = self._classify_collatz_block(crest_val)
            
            # Our hypothesis is that primes are most likely to be F-blocks,
            # as these represent the expansionary "push" in the harmonic model.
            if block_type == "F":
                # We've found a high-probability candidate. Now test it.
                if self._is_prime(num):
                    return num # Success! We've hopped to the next prime.
                # If it's not prime, we continue hopping from this point.
            
            if num > self.max_prime:
                return None


    def run(self):
        """
        Runs the prime prediction engine.
        """
        print("="*80)
        print("  Prime Prediction Engine Initiated")
        print("  Traversing the Collatz harmonic lattice to predict prime locations...")
        print("="*80)
        
        report_interval = 1000
        
        while self.last_known_prime < self.max_prime:
            next_prime = self._phase_hop(self.last_known_prime)
            
            if next_prime is None:
                print("\nSearch limit reached.")
                break
                
            self.primes_found.append(next_prime)
            self.last_known_prime = next_prime
            
            if len(self.primes_found) % report_interval == 0:
                success_rate = len(self.primes_found) / self.candidates_tested
                print(f"  > Primes Found: {len(self.primes_found):,} | Last Prime Found: {self.last_known_prime:,} | Success Rate: {success_rate:.4f}")


        print("\n" + "="*80)
        print("  Prime Prediction Complete")
        print("="*80)
        success_rate = len(self.primes_found) / self.candidates_tested
        print(f"  Total Primes Found: {len(self.primes_found):,}")
        print(f"  Total Candidates Tested: {self.candidates_tested:,}")
        print(f"  Final Success Rate (Primes Found per Candidate Tested): {success_rate:.4f}")
        print("\n  Last 10 Primes Found:")
        for p in self.primes_found[-10:]:
            print(f"    {p:,}")
        print("="*80)

class DynamicHarmonicContext:
    """
    Tracks and dynamically updates the target C-H-F ratios to adapt
    to the meta-spiral's drift.
    """
    def __init__(self, size=1000):
        self.history = deque(maxlen=size)
        # Initial ratios based on our previous 10M run
        self.target_ratios = {"C": 0.259, "H": 0.241, "F": 0.500}


    def add(self, block_type):
        """Adds a new block and updates the target ratios if full."""
        self.history.append(block_type)
        if len(self.history) == self.history.maxlen:
            counts = Counter(self.history)
            total = len(self.history)
            self.target_ratios["C"] = counts["C"] / total
            self.target_ratios["H"] = counts["H"] / total
            self.target_ratios["F"] = counts["F"] / total


    def is_plausible(self, candidate_block_type, current_context):
        """
        Checks if a candidate fits the dynamically updated harmonic rhythm.
        """
        if not current_context: return True
        
        temp_counts = current_context.copy()
        temp_counts[candidate_block_type] += 1
        total = sum(temp_counts.values())
        
        current_ratio = temp_counts[candidate_block_type] / total
        target_ratio = self.target_ratios[candidate_block_type]
        
        # A plausible candidate doesn't push the local ratio too far
        # from the learned target ratio.
        if current_ratio > target_ratio * 1.2: # Allow 20% local deviation
            return False
            
        return True
