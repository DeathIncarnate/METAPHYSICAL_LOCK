    <script>
        // ========================================================================
        // Main Verification Logic
        // ========================================================================
        const verifyBtn = document.getElementById('verifyBtn');
        const privateKeyInput = document.getElementById('privateKey');
        const originalHashInput = document.getElementById('originalHash');
        const resultsContainer = document.getElementById('results');
        
        const publicKeyCompOutput = document.getElementById('publicKeyComp');
        const publicKeyUncompOutput = document.getElementById('publicKeyUncomp');
        const finalHashOutput = document.getElementById('finalHash');
        const statusOutput = document.getElementById('status');


        // Helper function to convert a hex string to a Uint8Array
        function hexToBytes(hex) {
            const bytes = new Uint8Array(hex.length / 2);
            for (let i = 0; i < bytes.length; i++) {
                bytes[i] = parseInt(hex.substr(i * 2, 2), 16);
            }
            return bytes;
        }


        // Helper function to convert an ArrayBuffer to a hex string
        function bufferToHex(buffer) {
            return [...new Uint8Array(buffer)]
                .map(b => b.toString(16).padStart(2, '0'))
                .join('');
        }


        verifyBtn.addEventListener('click', async () => {
            const privateKeyHex = privateKeyInput.value.trim();
            const originalHashHex = originalHashInput.value.trim();


            if (privateKeyHex.length !== 64 || !/^[0-9a-fA-F]+$/.test(privateKeyHex)) {
                alert("Invalid Private Key. Must be 64 hex characters.");
                return;
            }
             if (originalHashHex.length !== 64 || !/^[0-9a-fA-F]+$/.test(originalHashHex)) {
                alert("Invalid Original Hash. Must be 64 hex characters.");
                return;
            }


            try {
                // 1. Initialize the secp256k1 curve
                const ec = new elliptic.ec('secp256k1');


                // 2. Generate the key pair from the private key
                const keyPair = ec.keyFromPrivate(privateKeyHex, 'hex');


                // 3. Get the public key in both compressed and uncompressed formats
                const publicKeyCompressed = keyPair.getPublic(true, 'hex');
                const publicKeyUncompressed = keyPair.getPublic(false, 'hex');


                // 4. Hash the compressed public key using the Web Crypto API for SHA-256
                const publicKeyBytes = hexToBytes(publicKeyCompressed);
                const hashBuffer = await crypto.subtle.digest('SHA-256', publicKeyBytes);
                const finalHashHex = bufferToHex(hashBuffer);


                // 5. Display the results
                publicKeyCompOutput.textContent = publicKeyCompressed;
                publicKeyUncompOutput.textContent = publicKeyUncompressed;
                finalHashOutput.textContent = finalHashHex;


                // 6. Compare the final hash with the original and set status
                if (finalHashHex === originalHashHex) {
                    statusOutput.textContent = "SUCCESS: HASHES MATCH";
                    statusOutput.className = "rounded-lg p-3 font-mono text-sm text-center font-bold status-match";
                } else {
                    statusOutput.textContent = "FAILURE: HASHES DO NOT MATCH";
                    statusOutput.className = "rounded-lg p-3 font-mono text-sm text-center font-bold status-no-match";
                }


                resultsContainer.classList.remove('hidden');


            } catch (error) {
                console.error("Verification Error:", error);
                alert("An error occurred during verification. Check the console for details.");
            }
        });
    </script>
