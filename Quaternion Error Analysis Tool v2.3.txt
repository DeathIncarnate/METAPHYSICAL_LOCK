    <script>
        // ========================================================================
        // Non-Markovian Analysis Engine
        // ========================================================================
        const analysisEngine = {
            hexToBytes(hex) {
                const bytes = new Uint8Array(hex.length / 2);
                for (let i = 0; i < bytes.length; i++) {
                    bytes[i] = parseInt(hex.substr(i * 2, 2), 16);
                }
                return bytes;
            },
            bytesToHex(bytes) {
                return Array.from(bytes).map(b => b.toString(16).padStart(2, '0')).join('');
            },
            calculateErrorVector(bytes1, bytes2) {
                const errorBytes = new Uint8Array(32);
                for (let i = 0; i < 32; i++) {
                    errorBytes[i] = bytes1[i] ^ bytes2[i];
                }
                return errorBytes;
            },
            getHarmonicSignature(hashBytes) {
                const signature = [];
                // The length of hashBytes can be partial (e.g., 4, 8, 12... bytes)
                const numBlocks = hashBytes.length / 4;
                for (let i = 0; i < numBlocks; i++) {
                    const block = new DataView(hashBytes.buffer, hashBytes.byteOffset).getUint32(i * 4, false);
                    if (block === 0) {
                        signature.push('Z');
                        continue;
                    }
                    const v2 = (block & -block).toString(2).length - 1;
                    let type;
                    if (v2 === 1) type = 'F';
                    else if (v2 === 2) type = 'H';
                    else type = 'C';
                    signature.push(type);
                }
                return signature;
            },
            deriveCorrectionOperator(signature) {
                const PHI = (1 + Math.sqrt(5)) / 2;
                let angle = 0;
                let axis = [0, 0, 0];
                signature.forEach((type, i) => {
                    switch(type) {
                        case 'C':
                            angle += (Math.PI / 4) / PHI;
                            axis[0] += 1 / (i + 1);
                            break;
                        case 'H':
                            angle += (Math.PI / 4);
                            axis[1] += 1 / (i + 1);
                            break;
                        case 'F':
                            angle += (Math.PI / 4) * PHI;
                            axis[2] += 1 / (i + 1);
                            break;
                        case 'Z': break;
                    }
                });
                if (angle === 0) return { angle: 0, axis: [0, 0, 0] };
                const mag = Math.sqrt(axis[0]**2 + axis[1]**2 + axis[2]**2);
                axis = axis.map(v => v / mag);
                return { angle, axis };
            },


            // Main function to run the sequential, non-Markovian trace
            runTrace(derivedHex, originalHex) {
                if (derivedHex.length !== 64 || !/^[0-9a-fA-F]+$/.test(derivedHex) ||
                    originalHex.length !== 64 || !/^[0-9a-fA-F]+$/.test(originalHex)) {
                    return { error: "Invalid hash format. Both must be 64 hex characters." };
                }


                const derivedBytes = this.hexToBytes(derivedHex);
                const originalBytes = this.hexToBytes(originalHex);
                const fullErrorVector = this.calculateErrorVector(derivedBytes, originalBytes);
                
                const traceLog = [];


                // Iterate through the 8 blocks of the error vector
                for (let i = 1; i <= 8; i++) {
                    // Create a cumulative error vector for the current step
                    const cumulativeErrorBytes = fullErrorVector.slice(0, i * 4);
                    
                    const signature = this.getHarmonicSignature(cumulativeErrorBytes);
                    const operator = this.deriveCorrectionOperator(signature);


                    traceLog.push({
                        block: i,
                        signature: signature.join(' '),
                        angle: operator.angle.toFixed(4),
                        axis: `[${operator.axis.map(v => v.toFixed(2)).join(', ')}]`
                    });
                }


                return {
                    fullErrorVector: this.bytesToHex(fullErrorVector),
                    trace: traceLog
                };
            }
        };


        // ========================================================================
        // UI Event Handling
        // ========================================================================
        const analyzeBtn = document.getElementById('analyzeBtn');
        const derivedHashInput = document.getElementById('derivedHash');
        const originalHashInput = document.getElementById('originalHash');
        const resultsContainer = document.getElementById('results');
        const traceTableBody = document.getElementById('traceTableBody');


        analyzeBtn.addEventListener('click', () => {
            const derivedHex = derivedHashInput.value.trim();
            const originalHex = originalHashInput.value.trim();


            const result = analysisEngine.runTrace(derivedHex, originalHex);


            if (result.error) {
                alert(result.error);
                return;
            }


            document.getElementById('errorVector').textContent = result.fullErrorVector;
            
            // Populate the trace table
            traceTableBody.innerHTML = ''; // Clear previous results
            result.trace.forEach(step => {
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td class="text-center">${step.block}</td>
                    <td>${step.signature}</td>
                    <td>${step.angle}</td>
                    <td>${step.axis}</td>
                `;
                traceTableBody.appendChild(row);
            });


            resultsContainer.classList.remove('hidden');
        });
    </script>
