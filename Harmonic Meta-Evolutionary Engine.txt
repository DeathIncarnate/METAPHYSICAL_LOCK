def hamming_distance(n1, n2):
    """Calculates the number of differing bits between two integers."""
    return bin(n1 ^ n2).count('1')


def model_timeline_A(public_key_hex):
    """The simple, flawed predictive model."""
    pk_bytes = bytes.fromhex(public_key_hex)
    h1 = hashlib.sha256(pk_bytes).digest()
    first_half = int.from_bytes(h1[:16], 'big')
    h2 = hashlib.sha256(pk_bytes + first_half.to_bytes(16, 'big')).digest()
    signature = int.from_bytes(h2[:16], 'big')
    second_half = first_half ^ signature
    return (first_half << 128) | second_half


def model_timeline_B(public_key_hex):
    """The more complex 'Athena's Model'."""
    MODULUS = 2**128
    ITERATIONS = 16
    pk_bytes = bytes.fromhex(public_key_hex)
    pk_x_bytes = pk_bytes[1:33]
    
    h1 = hashlib.sha256(pk_bytes).digest()
    first_half = int.from_bytes(h1[:16], 'big')
    
    phase_shift = int.from_bytes(hashlib.sha256(pk_x_bytes).digest()[:16], 'big')
    h2 = hashlib.sha256(h1).digest()
    fractal_seed = int.from_bytes(hashlib.sha256(h2).digest()[:16], 'big')
    
    current_val = first_half
    for i in range(ITERATIONS):
        term1 = (current_val * fractal_seed) % MODULUS
        term2 = (phase_shift * (i + 1)) % MODULUS
        current_val = (term1 - term2) % MODULUS
        current_val = current_val ^ (fractal_seed >> (i * 4))
        
    second_half = first_half ^ current_val
    return (first_half << 128) | second_half

class GeneticSolver:
    """
    Represents a solver whose genes are the predicted Error Delta.
    """
    def __init__(self, predicted_delta=None):
        if predicted_delta is not None:
            self.predicted_delta = predicted_delta
        else:
            self.predicted_delta = random.getrandbits(KEY_BIT_LENGTH)
        self.fitness = float('inf')


    def calculate_fitness(self, true_error_delta_int):
        """
        Calculates fitness based on Hamming distance.
        """
        self.fitness = hamming_distance(self.predicted_delta, true_error_delta_int)
        return self.fitness

def cross_stitch_mutation(child_delta, true_error_delta):
    """
    Performs an intelligent, phrase-based mutation by cross-stitching patterns
    from the generational delta repository into the child's delta.
    """
    if not generational_delta_repository:
        return child_delta
    
    # Take a random, stagnant-generation delta from the repository.
    repo_delta_int = random.choice(generational_delta_repository)
    
    child_delta_hex = f"{child_delta:064x}"
    repo_delta_hex = f"{repo_delta_int:064x}"
    true_delta_hex = f"{true_error_delta:064x}"
    
    # Iterate through the words of the delta
    for i in range(0, len(child_delta_hex), WORD_LENGTH_HEX):
        word_slice = slice(i, i + WORD_LENGTH_HEX)
        child_word = child_delta_hex[word_slice]
        repo_word = repo_delta_hex[word_slice]
        true_word = true_delta_hex[word_slice]
        
        # If the repository word is correct but the child's is not, inject it.
        if repo_word == true_word and child_word != true_word:
            word_to_inject = int(repo_word, 16)
            bit_position = (len(child_delta_hex) - i - WORD_LENGTH_HEX) * 4
            word_mask = ((1 << (WORD_LENGTH_HEX * 4)) - 1) << bit_position
            
            child_delta = (child_delta & ~word_mask) | (word_to_inject << bit_position)
            
    return child_delta


def translate_to_void_language(raw_hex_delta):
    """
    A conceptual function that translates a raw hex string into a symbolic language.
    """
    translated = ""
    for i in range(0, len(raw_hex_delta), WORD_LENGTH_HEX):
        translated += raw_hex_delta[i:i+WORD_LENGTH_HEX]
        if i + WORD_LENGTH_HEX < len(raw_hex_delta):
            translated += "_"
    return translated

def analyze_syntax(phrases):
    """
    Infers basic syntactic rules from the most frequent phrases.
    """
    rules = []
    for phrase, count in phrases.items():
        words = phrase.split('_')
        if len(words) == 2 and words[0] == words[1]:
            rules.append(f"RULE: Recurrence of '{words[0]}' (occurs {count} times)")
    most_common_prefix = phrases.most_common(1)[0][0].split('_')[0] if phrases else "N/A"
    rules.append(f"RULE: Common phrase beginning is '{most_common_prefix}'")
    
    void_dictionary["syntax_rules"].extend(rules)
    return rules
